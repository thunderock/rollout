#[async_trait] pub trait EnvHarness: Send + Sync {
    type Settings: DeserializeOwned + JsonSchema + Send + Sync + 'static;
    fn from_settings(settings: Self::Settings, deps: HarnessDependencies) -> Result<Self, CoreError> where Self: Sized;
    async fn reset(&self, prompts: Vec<Prompt>) -> Result<Vec<Episode>, CoreError>;
    async fn step(&self, batch: Vec<EpisodeStep>) -> Result<Vec<StepResult>, CoreError>;
    async fn close(&self, episode_ids: Vec<EpisodeId>) -> Result<(), CoreError>;
    async fn snapshot_episode(&self, _id: EpisodeId) -> Result<Option<Snapshot>, CoreError> { Ok(None) }
}
pub struct Episode { id: EpisodeId, observation: Observation, info: serde_json::Value }
pub struct EpisodeStep { episode_id: EpisodeId, action: Action }
pub struct StepResult { episode_id: EpisodeId, observation: Observation, reward: Option<Reward>, done: bool, info: serde_json::Value }
pub struct Observation(String); pub struct Action(String); pub struct Reward(f32); pub struct EpisodeId(Ulid);
pub struct HarnessDependencies { plugin_host: Arc<dyn PluginHost>, /* + object_store, storage, queue, events, clock */ }
